```typescript
function calculateMonthOverMonthChange(current: number, previous: number): number {
  if (previous === 0) return 0;  // Avoid division by zero
  return ((current - previous) / previous) * 100;
}

// Examples:
calculateMonthOverMonthChange(205, 180) // Returns: 13.89 (13.89% increase)
calculateMonthOverMonthChange(51, 45)   // Returns: 13.33 (13.33% increase)
calculateMonthOverMonthChange(100, 120) // Returns: -16.67 (16.67% decrease)
calculateMonthOverMonthChange(50, 0)    // Returns: 0 (no comparison possible)
```
